void reverse(int[] arr, int start, int end) {
    while (start < end) {
        int temp = arr[start];
        arr[start] = arr[end];
        arr[end] = temp;

        end--;
    }
}

int maxSum(int[] arr, int n) {
    int arrSum = 0;
    int currVal = 0;

    for (int i = 0; i < n; i++) {
        arrSum += arr[i];
        currVal += i * arr[i];
    }

    int maxVal = currVal;

    for (int j = 1; j < n; j++) {
        currVal = currVal + arrSum - n * arr[n - j];
        maxVal = Math.max(maxVal, currVal);
    }

    return maxVal;
}

void fixArray(int[] A, int n) {
    for (int i = 0; i < n; i++) {
        if (A[i] != -1 && A[i] != i) {
            int x = A[i];

            while (A[x] != -1 && A[x] != x) {

                x = y;
