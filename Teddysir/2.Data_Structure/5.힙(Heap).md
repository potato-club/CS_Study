import java.io.*;
import java.util.*;

public class Solution {

    public static void main(String[] args) {

        int[] scoville = {1, 2, 3, 9, 10, 12};
        int K = 7;

        System.out.println(solution(scoville, K));

    }
    // ì„ì€ ìŒì‹ì˜ ìŠ¤ì½”ë¹Œ ì§€ìˆ˜ = ê°€ì¥ ë§µì§€ ì•Šì€ ìŒì‹ì˜ ìŠ¤ì½”ë¹Œ ì§€ìˆ˜ + (ë‘ ë²ˆì§¸ë¡œ ë§µì§€ ì•Šì€ ìŒì‹ì˜ ìŠ¤ì½”ë¹Œ ì§€ìˆ˜ * 2)

    public static int solution(int[] scoville, int K) {
        PriorityQueue<Integer> minHeap = new PriorityQueue<>(); // ìµœì†Œ í™ ìƒì„±

        if(scoville.length <= 1) {
            return -1;
        }

        for (int scove : scoville) {
            minHeap.add(scove);
        }


        int count = 0;
        while (minHeap.peek() < K) {
            
            if(minHeap.size() <= 1) {
                return -1;
            }

            int first = minHeap.poll();
            int second = minHeap.poll();

            int temp = first + (second * 2);

            minHeap.add(temp);
            count++;

        }

        return count;

    }
}
